% Hinweis: hyperref wird bereits in 00_packages.tex geladen.

% Link-Farben: stabile ETH-Slots aus der Index-Palette (Blau/Petrol),
% unabhängig von der aktiven Kapitelfarbe.
\hypersetup{colorlinks=true, linkcolor=ChapterColor0, urlcolor=ChapterColor4}

% Öffentliche GitHub-Release-Version unten links, Seitenzahl unten rechts
% (overlay, layout-neutral). \ZSFReleaseID wird im Release-Workflow gesetzt.
\AddToShipoutPictureFG{%
  \AtPageLowerLeft{%
    \put(2mm,1mm){%
      \makebox(0,0)[lb]{\sffamily\bfseries\fontsize{9.5pt}{9.5pt}\selectfont\color{black}\ZSFReleaseID}%
    }%
    \put(\dimexpr\paperwidth-2mm\relax,1mm){%
      \makebox(0,0)[rb]{\sffamily\bfseries\fontsize{9.5pt}{9.5pt}\selectfont\color{black}\thepage}%
    }%
  }%
}

% Enge Math-Display-Abstände (wie Analysis: keine grossen Lücken um Formeln)
\AtBeginDocument{
  \setlength{\abovedisplayskip}{\ZSFspaceXS}
  \setlength{\belowdisplayskip}{\ZSFspaceXS}
  \setlength{\abovedisplayshortskip}{\ZSFspaceXS}
  \setlength{\belowdisplayshortskip}{\ZSFspaceXS}
  \setlength{\mathsurround}{\ZSFspaceXS}
}

% Kompakte Listen (wie Analysis: kein überschüssiges Whitespace)
\setlist{nosep, itemsep=0pt, parsep=0pt, topsep=\ZSFspaceXS, partopsep=0pt}
\setlist[itemize]{leftmargin=3mm}
\setlist[enumerate]{leftmargin=4mm}

% Tabellen-Zellenabstand einheitlich
\setlength{\tabcolsep}{\ZSFspaceM}
\setlength{\extrarowheight}{\ZSFtextMinPad}
