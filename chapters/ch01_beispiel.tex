\begin{runintext}
  \ZSFkeyword*{runintext} ist der ruhige Fliesstext-Baustein zwischen
  strukturierten Elementen. Er übernimmt automatisch die Lesbarkeitsregeln
  für schmale Spalten.
\end{runintext}

\begin{defbox}[Injektive Abbildung\ZSFTitleTag{defbox}]
  Eine Abbildung $f\colon A\to B$ heisst
  \ZSFkeyword[Injektivitat@Injektivität]{injektiv}, wenn
  $f(x_1)=f(x_2)$ stets $x_1=x_2$ impliziert.
  \ZSFhl{Gleiche Bilder erzwingen gleiche Urbilder.}
\end{defbox}

\begin{statementbox}[Extremwertsatz\ZSFTitleTag{statementbox}]
  Jede stetige Funktion auf einem kompakten Intervall nimmt ein Maximum und
  ein Minimum an.
\end{statementbox}
