\begin{formulabox}[title={Grundmengen und Operatoren\ZSFTitleTag{formulabox}}]
  \textVorFormel{Basis-Makros:}
  \[
    x\in\R,\quad z\in\C,\quad n\in\N,\quad k\in\Z,\quad q\in\Q
  \]
  \formulasep
  \[
    \dd x,\quad \norm{\vect v},\quad \abs{x},\quad \sgn(x),\quad
    \grad f,\quad \divg \vect F,\quad \rot \vect F
  \]
  \formulasep
  \ZSFItemHeading{Automatische Summen und Limites}
  \[
    \ZSFsumAuto{i=1}{n} i,\quad \ZSFlimAuto{x\to0}\frac{\sin x}{x}
  \]
  \textNachFormel{Stack-/Side-Modus bleibt zentral umschaltbar.}
  \formulanote{Separator und Note gruppieren zusammengehörige Formeln.}
\end{formulabox}
\textNachBox{Nachtext bleibt optisch an die Formelbox gebunden.}

\begin{formulabox}[title={Semantische Verbindung}]
  \[
    \ZSFmhlA{a^2}+2\ZSFmhlB{ab}+\ZSFmhlD{b^2}
    =\ZSFmhlC{(a+b)^2}
  \]
  \formulasep
  \[
    \ZSFbraceunder{x^2+2x+1}{Ausgangsform}
    =\ZSFbraceover{(x+1)^2}{Ziel}
  \]
\end{formulabox}
