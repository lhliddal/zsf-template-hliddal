\SubsectionBar[sec:markers]{Scan-Anker im Inhalt}
\ZSFindex{Scan-Anker}
\ZSFindex{Fernverweis}

\begin{runintext}
  Ein \ZSFkeyword{Schlagwort} ist der primäre Scan-Anker. Die
  \ZSFhl{prüfungskritischste Aussage} darf zusätzlich hervorgehoben werden.
  \ZSFdanger{Vorzeichen prüfen} markiert eine lokale Gefahr.
  \ZSFconclusion{Das Resultat ist sofort auffindbar.}
\end{runintext}

\end{defbox}
